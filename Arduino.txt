void setup() {
 
  Serial.begin(9600); 
  RemoteXY_Init ();                          // Инициализация RemoteXY
  strcpy(RemoteXY.Temp, "");
  strcpy(RemoteXY.Humidity, "");
  strcpy(RemoteXY.MQ_Sensor, "");
  strcpy(RemoteXY.Rasstoyanie[11], "");
  strcpy(RemoteXY.IK_Datchik,"");
  strcpy(RemoteXY.FLAME, "");
  strcpy(RemoteXY.Door_Status,"");
 //================================================================
  
  expander.begin(0x20);                      // Инициализация расширителя
  lcd.init();                                // Инициализация LCD-дисплея
  lcd.backlight();                           // Включение подсветки
  lcd.setCursor(6, 1);                       // Установка курсора на положение 6 символ 2 строка
  lcd.print("DOOR CLOSE");                   // Печать на LCD-дисплей
  
  strcpy(RemoteXY.Door_Status, "CLOSED");
  
  matr.shutdown(0, false);                 // Выключение энергосберегающий режим
  matr.setIntensity(0, 8);                 // Установка яркости на среднее значение
  matr.clearDisplay(0);                    // Отчистка матрицы
  
  expander.pinMode(LED_PIN_RED, OUTPUT);  // Настройка пина красного LED на выход
  expander.pinMode(LED_PIN_GREEN, OUTPUT);// Настройка пина зеленого LED на выход

  expander.pinMode(FlamePin, INPUT);      // Настройка пина Датчика Пламени(на расширителе) на вход
  expander.pinMode(SoundPin, INPUT);      // Настройка пина Датчика звука (на расширителе) на вход
  expander.pinMode(ZummerPin,OUTPUT);    // Настройка пина Зуммера (на расширителе) на выход
  
  pinMode(photoPin,INPUT);              // Настройка пина фоторезистора на вход
  pinMode(TrigPin, OUTPUT);             // Настройка пина Trig (Датчик Расстояния) на выход
  pinMode(EchoPin, INPUT);              // Настройка пина Echo (Датчик Расстояния) на выход
  pinMode(IKPin,   INPUT);              // Настройка пина Датчика Растояния(на расширителе) на вход
  pinMode(DHTPIN,  INPUT);              // Настройка датчика температуры DHT11 на вход
  
  
  digitalWrite(IKPin, LOW);                // Подача LOW на пин ИК-Датчика (нет движения)
  expander.digitalWrite(FlamePin, HIGH);   // Подача HIGH на пин Датчика пламени (нет огня)
  expander.digitalWrite(SoundPin, HIGH);   // Подача LOW на пин Датчика звука (нет звука)
  expander.digitalWrite(LED_PIN_GREEN, HIGH);  //"зажигаем" зеленый LED
  
  dht.begin();                          // Запускаем датчик температуры и влажности DHT11
  SPI.begin();                          // Запускаем передачу по SPI шине
  mfrc522.PCD_Init();                   // Инициализация RFID-считывателя

  myservo.write(angleDoorClose);  //закрываем дверь
  CloseDoor();
}

void loop() {

  RemoteXY_Handler (); //Обработчик RemoteXY
  ButtonHandler();    //Обработчик кнопки у приложения RemoteXY
  RFIDFunc();         //Функция для RFID-считывателя
  LCDFunc();          //Функция для вывода темературы и ... на LCD-дисплей
  RasstFunc();        //Функция для датчика расстояния
  IKFunc();           //Функция для ИК-Датчика

  if ((!numberDoorShow)&&(!nightOn)&&(!flameOn)&&(!gasOn)&&(!soundOn))
  PotenFace();        //Функция для потенциометра и датчика расстояния
 
  FlameFunc();        //Функция для датчика огня
  PhotoFun();            //Функция для фоторезистора  
  SoundFunc();        //Функция для датчика звука
  noTone(ZummerPin);

  if(doorOn) ThreeTwoOne(); //если дверь открыта, то через 10 сек закрываем и на матрицу "3-2-1" 
}

//Читаем значение кнопки RemoteXY
void ButtonHandler()
{
  if (RemoteXY.Door != 0) OpenDoor();       // Если кнопка в приложении находится в положении opened - выполняем функцию открывания двери,
  else CloseDoor();                         // Если в положении closed - выполняем фунцкию закрывания двери
}
//--------------------------------------------------------------------------------------------------------------------------
//Выводим 3-2-1 на Матрицу и закрываем дверь
void ThreeTwoOne() { 
  numberDoorShow=true;
  int t=millis() - doorOnTime; 
  if(t>7000) 
    {
      runMatrix(number3); //Вывод цифры 3 на Матрицу 
      delay(1000);
      runMatrix(number2);  //Вывод цифры 2 на Матрицу
      delay(1000);
      runMatrix(number1);  //Вывод цифры 1 на Матрицу
      delay(1000);    
      CloseDoor();
      RemoteXY.Door = 0;  //Значение кнопки присваиваем 0(CLOSED) для параллельного открытия и закрытия с телефона и RFID
      doorOn=false;
      numberDoorShow=false;
    }          
}
//--------------------------------------------------------------------------------------------------------------------------
//Открыть дверь
void OpenDoor() {                           // Функция открывания двери

  if (myservo.read() != angleDoorOpen)      // Если дверь закрыта(состояние сервопривода примерно 110-120)
  {
    toneFunc(ZummerPin, 1500, 100);         // Звоним, если дверь открывается
    toneFunc(ZummerPin, 1500, 100);
    strcpy(RemoteXY.Door_Status, "OPENED"); // Выводим значение "OPENED" в приложение RemoteXY

  //  lowEnergy(0);                           //Выключаем матрицу 8*8 и LCD-дисплей
    
    myservo.attach(ServoPin);               // Подлючение сервопривода к пину ServoPin
    myservo.write(angleDoorOpen);           // Открываем дверь(записываем значение angleDoorOpen0 в сервопривод)
    delay(delayDoor);                       // Задержка delayDoor (в мс). Необходима для корректного открытия двери, т.к дальше отключаем сервопривод
    myservo.detach();                       // Отключаем сервопривод от пина ServoPin

  //  lowEnergy(1);                           //Включаем матрицу 8*8 и LCD-дисплей
    
    lcd.setCursor(6, 1);                    // Выводим на LCD-Дисплей DOOR OPEN
    lcd.print("DOOR OPEN!");
  
    doorOnTime = millis();                  //фиксируем время открытия дверей с начала работы программы
    doorOn = true;                          //флаг состояния двери = true
    
    expander.digitalWrite(LED_PIN_RED, HIGH); 
    expander.digitalWrite(LED_PIN_GREEN,LOW); 
  }
}

//--------------------------------------------------------------------------------------------------------------------------
//Закрыть дверь
void CloseDoor() {                           // Функция открывания двери
  
  if (myservo.read() == angleDoorOpen)     // Если дверь открыта(состояние сервопривода 60)
  {
    toneFunc(ZummerPin, 1500, 100);         // Звоним, если дверь открывается
    toneFunc(ZummerPin, 1500, 100);
    strcpy(RemoteXY.Door_Status, "CLOSED"); // Выводим значение "CLOSED" в приложение RemoteXY
    
  // lowEnergy(0);                           //Выключаем матрицу 8*8 и LCD-дисплей
    
    myservo.attach(ServoPin);               // Подлючение сервопривода к пину ServoPin
    myservo.write(angleDoorClose);          // Закрываем дверь(записываем значение angleDoorClose в сервопривод)
    delay(delayDoor);                             // Задержка в 250 мс(Необходима для корректного открытия двери, т.к дальше отключаем сервопривод)
    myservo.detach();                       // Отключаем сервопривод от пина ServoPin
    
  //  lowEnergy(1);                           //Включаем матрицу 8*8 и LCD-дисплей
    
    lcd.setCursor(6, 1);                    // Выводим на LCD-Дисплей DOOR CLOSE
    lcd.print("DOOR CLOSE");
    doorOn = false;                          //флаг состояния двери = false
   
    expander.digitalWrite(LED_PIN_RED, LOW); 
    expander.digitalWrite(LED_PIN_GREEN,HIGH); 
   
  }
}

// Функция "Энергосберегающий режим". Выключает дисплеи при open/close двери
void lowEnergy(byte m){
 if (m==0){
    
    matr.clearDisplay(0); 
    matr.shutdown(0, true);             // Включение энергосберегающего режима Матрицы 8*8 (выключение дисплея)
    lcd.noDisplay();                     //Выключение LCD-дисплея  
    lcd.setBacklight(false);             //выключить подсветку LCD  
  }
  else{
    matr.shutdown(0, false);             //Выключение энергосберегающего режима Матрицы 8*8 (включение дисплея)
    lcd.display();                       //Включение LCD-дисплея       
    lcd.setBacklight(true);               //включить подсветку LCD
  }
  
  delay(250);  
}

// Функция для вывода температуры,влажности и др. на LCD дисплей
void LCDFunc()                              
{
  int t = dht.readTemperature();            // Считываем температуру в переменную
  int h = dht.readHumidity();               // Считываем влажность в переменную
 
 // Проверка чтения данных
  if (isnan(h) || isnan(t)) {
    Serial.println("Failed to read from DHT sensor!");
    t=25;h=47;
    return;
  } 
  
  int pm = analogRead(MQPin);               // Считываем значение датчика MQ в переменную
  /*  
     преобразуем значение концентрации газа в строку  
    и помещаем ее сразу в поле MQ_Sensor структуры RemoteXY  
  */ 
  dtostrf(pm, 0, 0, RemoteXY.MQ_Sensor);    // Выводим значение датчика MQ в приложение RemoteXY
  dtostrf(t, 0, 0, RemoteXY.Temp);          // Выводим значение датчика температуры в приложение RemoteXY
  dtostrf(h, 0, 0, RemoteXY.Humidity);      // Выводим значение датчика влажность в приложение RemoteXY
  lcd.setCursor(0, 0);                      //Выводим на первую строчку LCD-дисплея значения температуры, влажности и значения MQ-сенсора
  lcd.print("+");
  lcd.setCursor(1, 0);
  lcd.print(t);
  lcd.setCursor(3, 0);

// Функция RFID-считывателя
void RFIDFunc() 
{
    
  // Reset the loop if no new card present on the sensor/reader. This saves the entire process when idle.
  if (!mfrc522.PICC_IsNewCardPresent()) {
    delay(50);
   // Serial.println("No card");
    return;
  }   

  if (!mfrc522.PICC_ReadCardSerial()){
    delay(50);
    return;
  }
    
  String content = "";
  for (byte i = 0; i < mfrc522.uid.size; i++)
  {
    content.concat(String(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " "));
    content.concat(String(mfrc522.uid.uidByte[i], HEX));
  }
  content.toUpperCase();
 // Serial.print("Tag UID:"); //Вывод в консоль UID считываемого устройства(в нашем случае брелок или карта)
 // Serial.println(content);

   if (content.substring(1) == "E7 FC AD 26" || content.substring(1) == "D2 07 A9 1B" ) // Если UID карты или брелка совпадает с заданным,то
  {
    if (myservo.read() != angleDoorOpen) //Если дверь закрыта, то открываем дверь
    {
      OpenDoor();
      RemoteXY.Door = 1; //Значение кнопки присваиваем 1(OPEN) для параллельного открытия и закрытия с телефона и RFID
    }
    else
    {
      CloseDoor();
      RemoteXY.Door = 0;  //Значение кнопки присваиваем 0(CLOSED) для параллельного открытия и закрытия с телефона и RFID
    }
  }
}

// Функция зуммера
void toneFunc(byte pin, uint16_t frequency, uint16_t duration) //Функция для Зуммера(Реализация стандартной функции tone)
{
  unsigned long startTime = millis();
  unsigned long halfPeriod = 1000000L / frequency / 2;
  expander.pinMode(pin, OUTPUT);
  while (millis() - startTime < duration)
  {
    expander.digitalWrite(pin, HIGH);
    delayMicroseconds(halfPeriod);
    expander.digitalWrite(pin, LOW);
    delayMicroseconds(halfPeriod);
  }
    expander.pinMode(pin, INPUT);
}

// Функция мигания красным и зеленым светодиодом со звуком
void alarmLed() {
  expander.digitalWrite(LED_PIN_RED, HIGH);   // включаем красный LED (HIGH is the voltage level)
  expander.digitalWrite(LED_PIN_GREEN, LOW);  // выключаем зеленый LED (LOW is the voltage level)
  toneFunc(ZummerPin, 500, 100); 
  delay(100);                                 // пауза в сек
  expander.digitalWrite(LED_PIN_RED, LOW);    // выключаем красный LED (LOW is the voltage level)                                
  expander.digitalWrite(LED_PIN_GREEN, HIGH); //включаем зеленый LED (HIGH is the voltage level)                                
  toneFunc(ZummerPin, 2500, 100); 
  delay(100);                                 // пауза в сек
  
}

//Функция для расчета расстояния и вывода его на LCD-дисплей и в приложение RemoteXY
void RasstFunc() {  
  digitalWrite(TrigPin, LOW);
  delayMicroseconds(5);
  digitalWrite(TrigPin, HIGH); // Теперь установим высокий уровень на пине Trig
  delayMicroseconds(10);
  digitalWrite(TrigPin, LOW);
  int duration = pulseIn(EchoPin, HIGH); //Считываем сигнал с датчика: высокий импульс,длительность которого равна времени (в микросекундах) от отправки пинга до получения его эхо-сигнала от объекта.
  int cm = (duration / 2) / 29.1;    // преобразуем время в расстояние
  dtostrf(cm, 0, 0, RemoteXY.Rasstoyanie);
  lcd.setCursor(0, 1);
  lcd.print(cm);
  lcd.print("cm");
  lcd.print(" ");
  delay(100);
}
